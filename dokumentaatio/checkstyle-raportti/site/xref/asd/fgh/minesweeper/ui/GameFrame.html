19  /**
20   * Frame to visually represent an instance of a Game.
21   *
22   * @author ahv
23   */
24  public class GameFrame extends Frame {
25  
26      private final Game game;
27      private final GameGrid[][] grid;
28      private final Label timeLabel;
29      private final Label mineLabel;
30      private final Main main;
31  
32      public GameFrame(Main main, Difficulty difficulty) throws HeadlessException {
33          addWindowListener(new WindowAdapter() {
34              @Override
35              public void windowClosing(WindowEvent e) {
36                  System.exit(0);
37              }
38          });
39          setTitle("Minsweeper");
40          setFont(new Font("Arial", Font.BOLD, 16));
41  
42          this.main = main;
43          this.game = new Game(difficulty);
44          GameSettings s = game.getSettings();
45          setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
46  
47          Panel topPanel = new Panel(new FlowLayout());
48          this.timeLabel = new Label("Time: 0");
49          this.mineLabel = new Label("Mines: " + s.getMines());
50          topPanel.add(timeLabel);
51          topPanel.add(mineLabel);
52          topPanel.setSize(20, 200);
53          add(topPanel);
54  
55          int w = s.getWidth();
56          int h = s.getHeight();
57          Panel minePanel = new Panel(new GridLayout(h, w, 1, 1));
58          minePanel.setBackground(Color.GRAY);
59          this.grid = new GameGrid[w][h];
60          int x, y;
61          for (int i = 0; i < w * h; i++) {
62              x = i % w;
63              y = i / w;
64              GameGrid g = new GameGrid(this, game, x, y);
65              this.grid[x][y] = g;
66              minePanel.add(g);
67          }
68          //minePanel.setSize(40 * w, 40 * h);
69          add(minePanel);
70  
71          setSize(32 * w, 40 * h + 20);
72          setLocationRelativeTo(null);
73          setResizable(false);
74          setVisible(true);
75      }
76  
77      // TODO: Sketchy method
78      /**
79       * Gets a snapshot from the game object and updates the look accordingly.
80       *
81       */
82      public void updateView() {
83          if (game.hasEnded()) {
84              handleGameEnd(); // TODO: Show entire board on end
85          }
86          int[][] snap = game.getBoardSnapshot();
87          for (int x = 0; x < snap.length; x++) {
88              for (int y = 0; y < snap[0].length; y++) {
89                  this.grid[x][y].setVisualState(snap[x][y]);
90              }
91          }
92          // TODO: This should update "live"
93          this.timeLabel.setText("Time: " + game.getElapsedTime());
94          validate();
95          repaint();
96      }
97  
98      // TODO: UI is handling score logic, tsk tsk
99      private void handleGameEnd() {
100         main.showEndScreen(game.getFinalScore());
101     }
102 }
