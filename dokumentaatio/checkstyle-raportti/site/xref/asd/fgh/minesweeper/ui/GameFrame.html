3   import asd.fgh.minesweeper.logic.Difficulty;
4   import asd.fgh.minesweeper.logic.Game;

6   import java.awt.FlowLayout;
7   import java.awt.Frame;
8   import java.awt.GridLayout;
9   import java.awt.HeadlessException;
10  import java.awt.Label;
11  import java.awt.Panel;
12  import javax.swing.BoxLayout;
13  
14  /**
15   * Frame to visually represent an instance of a Game.
16   * @author ahv
17   */
18  public class GameFrame extends Frame {
19  
20      private final Game game;
21      private final GameButton[][] grid;
22      private final Label timeLabel;
23      private final Label mineLabel;
24      private final Main main;
25  
26      public GameFrame(Main main, Difficulty difficulty) throws HeadlessException {
27          this.main = main;
28          this.game = new Game(difficulty);
29          GameSettings s = game.getSettings();
30          setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
31  
32          Panel topPanel = new Panel(new FlowLayout());
33          this.timeLabel = new Label("Time: 0");
34          this.mineLabel = new Label("Mines: " + s.getMines());
35          topPanel.add(timeLabel);
36          topPanel.add(mineLabel);
37          topPanel.setSize(20, 200);
38          add(topPanel);
39  
40          int w = s.getWidth();
41          int h = s.getHeight();
42          Panel minePanel = new Panel(new GridLayout(w, h));
43          this.grid = new GameButton[w][h];
44          int x, y;
45          for (int i = 0; i < w * h; i++) {
46              x = i % w;
47              y = i / w;
48              // TODO: Seems pretty badly coupled but maybe we'll live.
49              GameButton b = new GameButton(this, game, x, y);
50              b.addMouseListener(b);
51              this.grid[x][y] = b;
52              minePanel.add(b);
53          }
54          minePanel.setSize(40 * h, 40 * w);
55          add(minePanel);
56  
57          setSize(40 * h + 20, 40 * w);
58          setLocationRelativeTo(null);
59          setVisible(true);
60      }
61  
62      // TODO: Sketchy method
63      /**
64       * Gets a snapshot from the game object and updates the look accordingly.
65       * @throws Exception 
66       */
67      public void updateView() {
68          if (game.hasEnded()) {
69              main.startGame(Difficulty.BEGINNER);
70          }
71          int[][] snap = game.getBoardSnapshot();
72          for (int x = 0; x < snap.length; x++) {
73              for (int y = 0; y < snap[0].length; y++) {
74                  this.grid[x][y].setVisualState(snap[x][y]);
75              }
76          }
77      }
78  }
