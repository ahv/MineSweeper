3   import java.awt.Color;

5   /**
6    * Represents a "visual" state of a grid.
7    *
8    * @author ahv
9    */
10  public enum GridState { // TODO: Iffy
11      NONE, ONE, TWO, THREE, FOUR, FIVE, SIX, SEVEN, EIGHT, MINED, FLAGGED, UNREVEALED;
12  
13      static GridState nthEnum(int n) { // a little unsafe perhaps.
14          return GridState.values()[n];
15      }
16  
17      /**
18       * Turns a GridState to a char, convenience for user interfaces.
19       *
20       * @param state The state to get a symbol for.
21       * @return An unique char matching each state.
22       */
23      public static char stateToSymbol(GridState state) {
24          switch (state) {
25              case NONE:
26              case UNREVEALED:
27                  return ' ';
28              case ONE:
29                  return '1';
30              case TWO:
31                  return '2';
32              case THREE:
33                  return '3';
34              case FOUR:
35                  return '4';
36              case FIVE:
37                  return '5';
38              case SIX:
39                  return '6';
40              case SEVEN:
41                  return '7';
42              case EIGHT:
43                  return '8';
44              case FLAGGED:
45                  return '!';
46              case MINED:
47                  return 'X';
48              default:
49                  return '?';
50          }
51      }
52  
53      static Color blue = new Color(0, 0, 220);
54      static Color darkblue = new Color(0, 0, 120);
55      static Color green = new Color(0, 120, 0);
56      static Color red = new Color(220, 0, 0);
57      static Color darkred = new Color(120, 0, 0);
58      static Color teal = new Color(0, 220, 220);
59      static Color purple = new Color(120, 0, 120);
60  
61      /**
62       * Convenience for user interfaces, matches states with colors.
63       *
64       * @param state The state to get a color for.
65       * @return A color matching a state.
66       */
67      public static Color stateToColor(GridState state) {
68          switch (state) {
69              case ONE:
70                  return blue;
71              case TWO:
72                  return green;
73              case THREE:
74                  return red;
75              case FOUR:
76                  return darkblue;
77              case FIVE:
78                  return darkred;
79              case SIX:
80                  return teal;
81              case SEVEN:
82                  return purple;
83              case NONE:
84              case UNREVEALED:
85              case EIGHT:
86              case FLAGGED:
87              case MINED:
88              default:
89                  return Color.black;
90          }
91      }
