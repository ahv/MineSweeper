1   package asd.fgh.minesweeper.logic;
2   
3   /**
4    * Time keeping object to score a game.
5    *
6    * @author ahv
7    */
8   public class StopWatch {
9   
10      private long startTime;
11      private long stopTime;
12      private boolean running;
13  
14      /**
15       * Start the stopwatch.
16       */
17      public void start() {
18          startTime = System.currentTimeMillis();
19          running = true;
20      }
21  
22      /**
23       * Stop the stopwatch.
24       */
25      public void stop() {
26          stopTime = System.currentTimeMillis();
27          running = false;
28      }
29  
30      /**
31       * Gives the elapsed time since first move in seconds.
32       *
33       * @return Elapsed time from first move, 0 if hasn't been started yet.
34       */
35      public int getElapsedTime() {
36          // long defaults to 0; means that watch hasn't been started
37          // unless system millis happen to be 0 when clock gets started, then whoops.
38          if (startTime == 0) {
39              return 0;
40          }
41          if (running) {
42              return (int) (System.currentTimeMillis() - startTime) / 1000;
43          }
44          return (int) (stopTime - startTime) / 1000;
45      }
46  }
